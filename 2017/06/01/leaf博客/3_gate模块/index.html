leaf是基于go语言实现开源网游服务器架构,见github.com/name5566/leaf

gate模块解析

gate意为网关，是整个服务器任务调度的核心功能之一，用于将服务收到的请求派发给相应处理流程单元(agent)
gate是Module一个实现，在gate初始化时，会实例化一个websocket Server或者tcp server，服务器监听相应的端口，在连接到来时，将连接派发给新建立的Agent对象。

调用流程图

Gate

  • leaf默认支持websocket和tcp服务，两者处理逻辑类似，就先看tcp的完整流程
type Gate struct {
    MaxConnNum      int 
    PendingWriteNum int
    MaxMsgLen       uint32
    Processor       network.Processor//提供数据输出的核心逻辑
    AgentChanRPC    *chanrpc.Server  //用于进行消息传递

    // websocket
    WSAddr      string
    HTTPTimeout time.Duration
    CertFile    string
    KeyFile     string

    // tcp
    TCPAddr      string
    LenMsgLen    int
    LittleEndian bool
}

gate实例化TCPServer的流程:
可以看到TCPSever初始化的同时，获得了Agent实例化的接口，这样在Server内部即可完成每个请求处理的完成生命周期

func (gate *Gate) Run(closeSig chan bool) {
    var tcpServer *network.TCPServer
    if gate.TCPAddr != "" {
        tcpServer = new(network.TCPServer)
        tcpServer.Addr = gate.TCPAddr
        tcpServer.MaxConnNum = gate.MaxConnNum
        tcpServer.PendingWriteNum = gate.PendingWriteNum
        tcpServer.LenMsgLen = gate.LenMsgLen
        tcpServer.MaxMsgLen = gate.MaxMsgLen
        tcpServer.LittleEndian = gate.LittleEndian
        tcpServer.NewAgent = func(conn *network.TCPConn) network.Agent {
            a := &agent{conn: conn, gate: gate}
            if gate.AgentChanRPC != nil {
                gate.AgentChanRPC.Go("NewAgent", a)
            }
            golog.Println("gate Agent new")
            return a
        }
    }

    if tcpServer != nil {
        tcpServer.Start()
    }
    <-closeSig
    if tcpServer != nil {
        tcpServer.Close()
    }
}

func (server *TCPServer) Start() {
    server.init()//初始化就是端口的监听
    go server.run()
}

//2. server处理步骤
func (server *TCPServer) run() {
    for {
        //a. 连接获取
        conn, err := server.ln.Accept()

        ...略

        //b. 实例化agent
        tcpConn := newTCPConn(conn, server.PendingWriteNum, server.msgParser)
        agent := server.NewAgent(tcpConn)

        //c. 调用agent完成整个消息处理
        go func() {
            agent.Run()
            ...略
        }()
    }
}

//3. 关闭监听，释放连接
func (server *TCPServer) Close() {
    server.ln.Close()
    server.wgLn.Wait()

    server.mutexConns.Lock()
    for conn := range server.conns {
        conn.Close()
    }
    server.conns = nil
    server.mutexConns.Unlock()
    server.wgConns.Wait()
}
